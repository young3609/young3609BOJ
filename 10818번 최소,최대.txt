#define _CRT_SECURE_NO_WARNINGS
#include <cstdio>
using namespace std;

int main() {

	int testcase;
	scanf("%d",&testcase);

	int max = -1000001;
	int min = 1000001;

	for (int i = 0; i < testcase; i++) {

		int n;
		scanf("%d",&n);

		if (n > max) { max = n; }
		if (n < min) { min = n; }

	}

	printf("%d %d\n",min,max);
	return 0;
}
