#include <iostream>
#include <string>
using namespace std;
#define div 1000000;
long long d[5001];

int main() {

	string s;
	cin >> s;

	int n = s.length();
	s = " " + s;
	d[0] = 1;
	for (int i = 1; i <= n; i++) {
		int x = s[i] - '0';
		if (1 <= x && x <= 9) {
			d[i] = (d[i] + d[i - 1]) % div;
		}
		if (i == 1)continue;
		if (s[i - 1] == '0')continue;
		x = (s[i - 1] - '0') * 10 + (s[i] - '0');
		if (10 <= x && x <= 26) {
			d[i] = (d[i] + d[i - 2]) % div;
		}
	}
	cout << d[n] << '\n';
	return 0;
}
