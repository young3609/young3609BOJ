#define _CRT_SECURE_NO_WARNINGS
#include <cstdio>
#include <algorithm>
using namespace std;

int d[1001];
int b[1001];
int a[1001];

int main() {

	int n;
	scanf("%d",&n);

	for (int i = 0; i < n; i++) {
		scanf("%d",&b[i]);
	}
	for (int i = 0; i < n; i++) {
		a[i] = b[n - i-1];
	}
	for (int i = 0; i < n; i++) {
		d[i] = 1;
		for (int j = 0; j < i; j++) {
			if (a[j] < a[i] && d[i] < d[j] + 1) {
				d[i] = d[j] + 1;
			}
		}
	}
	int maxresult = 0;
	for (int i = 0; i < n; i++) {
		if (maxresult < d[i]) {
			maxresult = d[i];
		}
	}
	printf("%d\n",maxresult);
	return 0;
}
