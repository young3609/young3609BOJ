// 规过 1

#define _CRT_SECURE_NO_WARNINGS
#include <cstdio>
using namespace std;

long long d[91];

int main() {

	int n;
	scanf("%d",&n);

	d[1] = 1;
	d[2] = 1;
	for (int i = 3; i <= n; i++) {
		d[i] = d[i - 1] + d[i - 2];
	}
	
	printf("%lld\n",d[n]);
	return 0;
}




// 规过 2

#define _CRT_SECURE_NO_WARNINGS
#include <cstdio>
using namespace std;

long long dp[91][3];

int main() {

	int n;
	scanf("%d",&n);

	dp[1][1] = 1;
	dp[2][0] = 1;

	for (int i = 3; i <= n; i++) {
		dp[i][0] = dp[i - 1][0] + dp[i - 1][1];
		dp[i][1] = dp[i - 1][0];
	}

	printf("%lld\n",dp[n][0]+dp[n][1]);
	return 0;
}
